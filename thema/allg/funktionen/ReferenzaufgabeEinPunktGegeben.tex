
\subsubsection{Ein Punkt ist gegeben}\index{Punkt auf Geraden}\index{Gerade!Punkt auf|textbf}
Bei vielen Anwendungen ist von der Geraden die Steigung $a$
\textbf{oder} der $y$-Ach\-sen\-ab\-schnitt $b$ gegeben, aber nicht
beides. Dabei ist meist ein Punkt $P$ (\zB $P=(7|4)$) gegeben, durch
den die Gerade laufen muss.

Gleich ein Beispiel:

\begin{tabular}{p{12cm}}\hline
   $y$-Achsenabschnitt gegeben \\\hline
   Steigung gesucht \\\hline
  ${\color{orange}y}=a\cdot{}{\color{blue}x}+1.5$\\\hline
  Punkt $P=({\color{blue}7}|{\color{orange}4})$ gegeben\\\hline
  einsetzen: \\
   $\LoesungsRaumKurz{{\color{orange}4}}=a\cdot{}\LoesungsRaumKurz{{\color{blue}7}} + 1.5$\\\hline
 lösen\\
  \end{tabular}


\TNT{6}{
\begin{tabular}{c}
 $ 4           =7a +1.5$ \\
 $ 2.5         =7a     $ \\
 $ \frac{2.5}7 = a     $ \\
 $ a = \frac{5}{14} \approx 0.357    $ \\
 $ y=\frac5{14}\cdot{x} + 1.5   $ \\
\end{tabular}
}

\begin{rezept}{Einsetzen}{}
Koordinaten des gegebenen Punktes in die
  Geradengleichung einsetzen, um $a$ bzw. $b$ zu finden!
\end{rezept}
