\subsubsection{Vorzeigeaufgabe}
Buch S. 118 Aufg. 364. c):


$$a^{x+1} - a^{x-1} = b^x$$
\TNT{10}{%%
  $$a^x\cdot{}a - a^x\cdot{}\frac1a = b^x$$
  $$a^x \left(a-\frac1a\right) = b^x$$
  $$a^x \left(\frac{a^2-1}a\right) = b^x$$
  $$\frac{a^x}{b^x} = \frac{a}{a^2 - 1}$$
  $$\left(\frac{a}{b}\right)^x = \frac{a}{a^2 - 1}$$
  $$x\cdot{} \ln\left(\frac{a}b\right) = \ln\left(\frac{a}{a^2-1}\right)$$
  $$x = \frac{\ln\left(\frac{a}{a^2-1}\right)}{\ln\left(\frac{a}b\right)}$$
}%% END TNT
